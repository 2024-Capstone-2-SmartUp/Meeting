```
import io
import os
from google.cloud import speech
from pydub import AudioSegment
import openai  # OpenAI 패키지

# Google Cloud 서비스 계정 키 파일 경로 설정
os.environ["GOOGLE_APPLICATION_CREDENTIALS"] = "C:/Users/halla/Downloads/capstone.json"

def transcribe_audio(audio_file_path, output_file_path):
    # Speech-to-Text 클라이언트 생성
    client = speech.SpeechClient()

    # 오디오 파일 불러오기 (wav 형식으로 가정)
    sound = AudioSegment.from_wav(audio_file_path)

    # 스테레오 오디오를 모노로 변환
    sound = sound.set_channels(1)

    # 오디오 파일을 더 작은 크기로 분할 (예: 30초)
    chunk_length_ms = 30000  # 30초
    chunks = [sound[i:i + chunk_length_ms] for i in range(0, len(sound), chunk_length_ms)]

    # 결과 텍스트 저장할 파일 열기
    with open(output_file_path, "w", encoding="utf-8") as output_file:
        for i, chunk in enumerate(chunks):
            # 각 chunk를 임시 파일로 저장
            chunk_file_path = f"temp_chunk_{i}.wav"
            chunk.export(chunk_file_path, format="wav")

            # 음성 파일을 열고 데이터를 읽음
            with io.open(chunk_file_path, "rb") as audio_file:
                content = audio_file.read()

            # 음성 파일 정보 설정
            audio = speech.RecognitionAudio(content=content)
            config = speech.RecognitionConfig(
                encoding=speech.RecognitionConfig.AudioEncoding.LINEAR16,
                sample_rate_hertz=44100,  # 실제 샘플 레이트와 맞춤
                language_code="ko-KR"  # 한국어 설정
            )

            # 음성 파일을 텍스트로 변환 요청
            response = client.recognize(config=config, audio=audio)

            # 결과 출력 및 파일에 저장
            for result in response.results:
                transcript = result.alternatives[0].transcript
                print(f"Chunk {i + 1} Transcribed Text: {transcript}")
                output_file.write(f"Chunk {i + 1}: {transcript}\n")

            # 임시 파일 삭제
            os.remove(chunk_file_path)

    # STT로 변환된 텍스트 파일 읽기
    with open(output_file_path, "r", encoding="utf-8") as f:
        transcript_text = f.read()

    return transcript_text

# 텍스트를 n개의 조각으로 나누는 함수
def split_text_into_chunks(text, num_chunks):
    chunk_size = len(text) // num_chunks
    chunks = []
    start = 0

    for i in range(num_chunks - 1):
        end = text.rfind('\n', start, start + chunk_size)
        if end == -1:
            end = start + chunk_size
        chunks.append(text[start:end].strip())
        start = end

    chunks.append(text[start:].strip())  # 마지막 조각 추가
    return chunks

# GPT API 요약 및 결과 저장 (조각별로 나눠서 요약)
def summarize_text_in_chunks(text, output_summary_file, lecture_duration_hours):
    openai.api_key = 'OpenAi Key'

    # 강의 길이에 따른 조각 수 계산
    if lecture_duration_hours == 1:
        num_chunks = 2
    elif lecture_duration_hours == 2:
        num_chunks = 4
    elif lecture_duration_hours == 3:
        num_chunks = 6
    else:
        num_chunks = max(2, lecture_duration_hours * 2)  # 예외 처리: 시간이 1, 2, 3시간이 아닐 경우

    # 텍스트를 조각으로 나누기
    chunks = split_text_into_chunks(text, num_chunks)

    # 요약된 텍스트를 파일에 저장
    with open(output_summary_file, "w", encoding="utf-8") as summary_file:
        for i, chunk in enumerate(chunks):
            messages = [{'role': 'user', 'content': '이 텍스트를 요약해줘: ' + chunk}]
            completion = openai.ChatCompletion.create(
                model='gpt-3.5-turbo',
                messages=messages
            )
            summary = completion['choices'][0]['message']['content']
            summary_file.write(f"Chunk {i + 1} Summary:\n{summary}\n\n")

    print(f"요약된 텍스트가 {output_summary_file}에 저장되었습니다.")

# 사용할 음성 파일 경로와 출력 파일 경로 설정
audio_file_path = "C:/studyC/test.wav"  # 실제 음성 파일 경로로 대체
output_file_path = "C:/studyC/transcribed_text.txt"  # 변환된 텍스트 저장 경로
output_summary_file = "C:/studyC/summary.txt"  # 요약된 텍스트 저장 경로
lecture_duration_hours = 2  # 강의 시간 (2시간 또는 3시간으로 설정 가능)

# 음성 파일을 STT로 변환하고 결과 요약
transcribed_text = transcribe_audio(audio_file_path, output_file_path)
summarize_text_in_chunks(transcribed_text, output_summary_file, lecture_duration_hours)
```
